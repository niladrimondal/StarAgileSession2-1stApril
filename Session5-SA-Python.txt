class Calculator:
    def __init__(self,num1,num2):
        self.a = num1
        self.b = num2


    def sum(self):
        return self.a + self.b
    
print('Enter the First Number')
num1 = int(input())

print('Enter the secone nuber')
num2 = int(input())

calc = Calculator(num1,num2)

print('sum of the two number is:')
print(calc.sum())
